} } class HomePage extends StatefulWidget { const HomePage({super.key}); @override State createState() => _HomePageState(); } class _HomePageState extends State { // 設定在一起的起點日期：2025 年 10 月 26 日 final DateTime _startDate = DateTime(2025, 10, 26); final List> _allMemories = [ {"date": "2026-05-20", "content": "一起去海邊看日落，風很溫柔，你也是。"}, {"date": "2026-05-14", "content": "紀念日大餐！那家餐酒館的松露薯條超好吃 🍟"}, {"date": "2026-05-01", "content": "解鎖了新技能：兩個人一起組裝完客廳的櫃子，累炸但很有成就感！"}, ]; List> _filteredMemories = []; bool _isSearching = false; final TextEditingController _dialogController = TextEditingController(); final TextEditingController _searchController = TextEditingController(); // 自動依日期排序 _dialogController.clear(); _selectedDate = DateTime.now(); // 儲存後重設搜尋狀態，避免畫面元素突兀消失 _resetSearch(); _runFilter(); }); Navigator.pop(context); } } void _deleteMemory(Map item) { showDialog( context: context, builder: (BuildContext context) { return AlertDialog( title: const Text('刪除回憶 😢'), content: const Text('確定要刪除這段甜蜜的回憶嗎？刪除後就找不回來囉！'), actions: [ TextButton( onPressed: () => Navigator.pop(context), child: const Text('取消', style: TextStyle(color: Colors.grey)), ), TextButton( onPressed: () { setState(() { _allMemories.remove(item); _resetSearch(); // 刪除後同樣重設搜尋狀態 _runFilter(); }); Navigator.pop(context); }, child: const Text('確定刪除', style: TextStyle(color: Colors.red)), ), ], ); }, ); } Future _pickDate(BuildContext context, StateSetter setModalState) async { final DateTime? picked = await showDatePicker( context: context, initialDate: _selectedDate, firstDate: DateTime(2000), lastDate: DateTime(2100), builder: (context, child) { return Theme( data: Theme.of(context).copyWith( colorScheme: const ColorScheme.light(primary: Colors.pink), ), child: child!, ); }, ); if (picked != null && picked != _selectedDate) { setModalState(() { _selectedDate = picked; }); } } void _showMemoryDialog({Map? originalItem}) { if (originalItem != null) { _dialogController.text = originalItem['content']!; _selectedDate = DateTime.parse(originalItem['date']!); } else { _dialogController.clear(); _selectedDate = DateTime.now(); } showModalBottomSheet( context: context, isScrollControlled: true, builder: (BuildContext context) { return StatefulBuilder( builder: (BuildContext context, StateSetter setModalState) { return Padding( padding: EdgeInsets.only( bottom: MediaQuery.of(context).viewInsets.bottom, left: 20, right: 20, top: 20, ), child: Column( mainAxisSize: MainAxisSize.min, children: [ Text( originalItem != null ? '修改我們的故事 ✍️' : '紀錄我們的新瞬間 ✨', style: const TextStyle(fontSize: 18, fontWeight: FontWeight.bold, color: Colors.pink), ), const SizedBox(height: 15), Row( mainAxisAlignment: MainAxisAlignment.center, children: [ Text( "日期：${_selectedDate.toString().split(' ')[0]}", style: const TextStyle(fontSize: 16, fontWeight: FontWeight.w500), ), const SizedBox(width: 10), TextButton.icon( onPressed: () => _pickDate(context, setModalState), icon: const Icon(Icons.calendar_month, color: Colors.pink), label: const Text('更換日子', style: TextStyle(color: Colors.pink)), ), ], ), const SizedBox(height: 10), TextField( controller: _dialogController, maxLines: 3, decoration: const InputDecoration( hintText: "寫下此刻的心情...", border: OutlineInputBorder(), ), ), const SizedBox(height: 15), ElevatedButton( onPressed: () => _saveMemory(originalItem: originalItem), style: ElevatedButton.styleFrom( backgroundColor: Colors.pink.withAlpha(26), // 修正 M3 的寫法 ), child: const Text('儲存回憶'), ), const SizedBox(height: 20), ], ), ); }, ); }, ); } @override Widget build(BuildContext context) { return Scaffold( appBar: AppBar( title: const Text('Our Times'), actions: [ IconButton( icon: Icon(_isSearching ? Icons.close : Icons.search), onPressed: () { setState(() { _isSearching = !_isSearching; if (!_isSearching) { _searchController.clear(); } }); }, ) ], ), body: Column( children: [ if (_isSearching) Padding( padding: const EdgeInsets.all(8.0), child: TextField( controller: _searchController, decoration: const InputDecoration( labelText: '搜尋回憶', prefixIcon: Icon(Icons.search), border: OutlineInputBorder(), ), ), ), Padding( padding: const EdgeInsets.all(16.0), child: Card( color: Colors.pink.shade50, child: Padding( padding: const EdgeInsets.all(20.0), child: Row( mainAxisAlignment: MainAxisAlignment.center, children: [ const Icon(Icons.favorite, color: Colors.pink), const SizedBox(width: 10), Text( '我們在一起的第 ${_calculateDaysTogether()} 天', style: const TextStyle(fontSize: 18, fontWeight: FontWeight.bold, color: Colors.pink), ), ], ), ), ), ), Expanded( child: _filteredMemories.isEmpty ? const Center(child: Text('目前沒有找到回憶唷 📝', style: TextStyle(color: Colors.grey))) : ListView.builder( itemCount: _filteredMemories.length, itemBuilder: (context, index) { final memory = _filteredMemories[index]; return Card( margin: const EdgeInsets.symmetric(horizontal: 16, vertical: 8), child: ListTile( title: Text(memory['content']!), subtitle: Text(memory['date']!), trailing: Row( mainAxisSize: MainAxisSize.min, children: [ IconButton( icon: const Icon(Icons.edit, color: Colors.blue), onPressed: () => _showMemoryDialog(originalItem: memory), ), IconButton( icon: const Icon(Icons.delete, color: Colors.red), onPressed: () => _deleteMemory(memory), ), ], ), ), ); }, ), ), ], ), floatingActionButton: FloatingActionButton( onPressed: () => _showMemoryDialog(), backgroundColor: Colors.pink, child: const Icon(Icons.add, color: Colors.white), ), ); } }
